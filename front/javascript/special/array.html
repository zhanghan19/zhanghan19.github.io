    // 移动数组中的内容
function moveArray(array, fro, to){
    if (fro < 0 || to >= array.length) {
        throw Error('参数错误')
        return
    }
    const newArray = [...array];
    let item = newArray.splice(fro, 1);
    newArray.splice(to, 0, ...item);
    return newArray;
}
